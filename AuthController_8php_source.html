310 private function generateJWT($header, $payload) {

324 public function verifyToken($jwt) {
325 try {
326 // Verificar si el token existe en la base de datos
327 $tokenData = $this->db->query(
328 "SELECT * FROM users WHERE jwt = ?",
329 [$jwt]
330 )->fetch();
331
332 if (!$tokenData) {
333 return false;
334 }
335
336 // Verificar estructura del token
337 list($header_encoded, $payload_encoded, $signature_encoded) = explode('.', $jwt);
338
339 // Verificar firma
340 $signature = hash_hmac('sha256', "$header_encoded.$payload_encoded", $this->secretKey, true);
341 $signature_check = $this->base64URLEncode($signature);
342
343 if ($signature_encoded !== $signature_check) {
344 return false;
345 }
346
347 // Verificar expiración
348 $payload = json_decode(base64_decode($payload_encoded), true);
349 if (isset($payload['exp']) && $payload['exp'] < time()) {
350 return false; // Expirado
351 }
352
353 return $payload;
354 } catch (\Exception $e) {
355 return false;
