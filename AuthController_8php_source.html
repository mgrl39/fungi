231 private function generateJWT($header, $payload) {
232 // Codificar header y payload
233 $header_encoded = $this->base64URLEncode(json_encode($header));
234 $payload_encoded = $this->base64URLEncode(json_encode($payload));

250 public function verifyToken($jwt) {
251 try {
252 // Verificar si el token existe en la base de datos
253 $tokenData = $this->db->query(
254 "SELECT * FROM users WHERE jwt = ?",
255 [$jwt]
256 )->fetch();
257
258 if (!$tokenData) {
259 return false;
260 }
261
262 // Verificar estructura del token
263 list($header_encoded, $payload_encoded, $signature_encoded) = explode('.', $jwt);
264
265 // Verificar firma
266 $signature = hash_hmac('sha256', "$header_encoded.$payload_encoded", $this->secretKey, true);
267 $signature_check = $this->base64URLEncode($signature);
268
269 if ($signature_encoded !== $signature_check) {
270 return false;
271 }
272
273 // Verificar expiración
274 $payload = json_decode(base64_decode($payload_encoded), true);
275 if (isset($payload['exp']) && $payload['exp'] < time()) {
276 return false; // Expirado
277 }
278
279 return $payload;
280 } catch (\Exception $e) {
281 return false;
282 }
