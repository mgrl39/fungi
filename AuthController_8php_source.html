309 private function generateJWT($header, $payload) {
310 // Codificar header y payload
311 $header_encoded = $this->base64URLEncode(json_encode($header));
312 $payload_encoded = $this->base64URLEncode(json_encode($payload));

328 public function verifyToken($jwt) {
329 try {
330 // Verificar si el token existe en la base de datos
331 $tokenData = $this->db->query(
332 "SELECT * FROM users WHERE jwt = ?",
333 [$jwt]
334 )->fetch();
335
336 if (!$tokenData) {
337 return false;
338 }
339
340 // Verificar estructura del token
341 list($header_encoded, $payload_encoded, $signature_encoded) = explode('.', $jwt);
342
343 // Verificar firma
344 $signature = hash_hmac('sha256', "$header_encoded.$payload_encoded", $this->secretKey, true);
345 $signature_check = $this->base64URLEncode($signature);
346
347 if ($signature_encoded !== $signature_check) {
348 return false;
349 }
350
351 // Verificar expiración
352 $payload = json_decode(base64_decode($payload_encoded), true);
353 if (isset($payload['exp']) && $payload['exp'] < time()) {
354 return false; // Expirado
355 }

357 return $payload;
358 } catch (\Exception $e) {
359 return false;
