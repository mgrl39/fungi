301 private function generateJWT($header, $payload) {

315 public function verifyToken($jwt) {
316 try {
317 // Verificar si el token existe en la base de datos
318 $tokenData = $this->db->query("SELECT * FROM users WHERE jwt = ?", [$jwt])->fetch();
319 if (!$tokenData) return false;
320 // Verificar estructura del token
321 list($header_encoded, $payload_encoded, $signature_encoded) = explode('.', $jwt);
322 // Verificar firma
323 $signature = hash_hmac('sha256', "$header_encoded.$payload_encoded", $this->secretKey, true);
324 $signature_check = $this->base64URLEncode($signature);
325 if ($signature_encoded !== $signature_check) return false;
326 // Verificar expiración
327 $payload = json_decode(base64_decode($payload_encoded), true);
328 if (isset($payload['exp']) && $payload['exp'] < time()) return false; // Expirado
329 return $payload;
330 } catch (\Exception $e) {
331 return false;
332 }
