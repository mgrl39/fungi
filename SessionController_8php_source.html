160 $tokenParts = explode('.', $jwt);
161
162 if (count($tokenParts) != 3) return false;
163 list($headerEncoded, $payloadEncoded, $signatureEncoded) = $tokenParts;
164 // Verificar firma
165 $signature = hash_hmac('sha256', "$headerEncoded.$payloadEncoded", $secretKey, true);
