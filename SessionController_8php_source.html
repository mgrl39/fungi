311 public function verifyJWT($jwt, $secretKey) {
312 $tokenParts = explode('.', $jwt);
313
314 if (count($tokenParts) != 3) {
315 return false;
316 }
317
318 list($headerEncoded, $payloadEncoded, $signatureEncoded) = $tokenParts;
319
320 // Verificar firma
321 $signature = hash_hmac('sha256', "$headerEncoded.$payloadEncoded", $secretKey, true);
322 $signatureCheck = $this->base64URLEncode($signature);
323
324 if ($signatureCheck !== $signatureEncoded) {
325 return false;
326 }
327
328 // Verificar expiración
329 $payload = json_decode($this->base64URLDecode($payloadEncoded), true);
330
331 if (!$payload || !isset($payload['exp'])) {
332 return false;
333 }
334
335 if ($payload['exp'] < time()) {
336 return false; // Token expirado
337 }
