123 public function verifyJWT($jwt, $secretKey) {
124 $tokenParts = explode('.', $jwt);
125
126 if (count($tokenParts) != 3) {
127 return false;
128 }
129
130 list($headerEncoded, $payloadEncoded, $signatureEncoded) = $tokenParts;
131
132 // Verificar firma
133 $signature = hash_hmac('sha256', "$headerEncoded.$payloadEncoded", $secretKey, true);
134 $signatureCheck = $this->base64URLEncode($signature);
135
136 if ($signatureCheck !== $signatureEncoded) {
137 return false;
138 }
139
140 // Verificar expiración
141 $payload = json_decode($this->base64URLDecode($payloadEncoded), true);
142
143 if (!$payload || !isset($payload['exp'])) {
144 return false;
145 }
146
147 if ($payload['exp'] < time()) {
148 return false; // Token expirado
149 }
