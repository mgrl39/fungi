177 public function verifyJWT($jwt, $secretKey) {
178 $tokenParts = explode('.', $jwt);
179
180 if (count($tokenParts) != 3) return false;
181 list($headerEncoded, $payloadEncoded, $signatureEncoded) = $tokenParts;
182 // Verificar firma
183 $signature = hash_hmac('sha256', "$headerEncoded.$payloadEncoded", $secretKey, true);
184 $signatureCheck = $this->base64URLEncode($signature);
185 if ($signatureCheck !== $signatureEncoded) return false;

187 // Verificar expiración
