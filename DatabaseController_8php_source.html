100 // Búsqueda avanzada de hongos por múltiples criterios
101 public function searchFungi($criteria = []) {
102 $query = "
103 SELECT f.*, t.*, c.*,
104 GROUP_CONCAT(DISTINCT CONCAT(ic.path, i.filename)) as image_urls
105 FROM fungi f
106 LEFT JOIN taxonomy t ON f.id = t.fungi_id
107 LEFT JOIN characteristics c ON f.id = c.fungi_id
108 LEFT JOIN fungi_images fi ON f.id = fi.fungi_id
109 LEFT JOIN images i ON fi.image_id = i.id
110 LEFT JOIN image_config ic ON i.config_key = ic.config_key
111 WHERE 1=1
112 ";
113 $params = [];
114
115 if (!empty($criteria['edibility'])) {
116 $query .= " AND f.edibility = :edibility";
117 $params[':edibility'] = $criteria['edibility'];
118 }
119 if (!empty($criteria['family'])) {
120 $query .= " AND t.family LIKE :family";
121 $params[':family'] = "%{$criteria['family']}%";
122 }
123 if (!empty($criteria['name'])) {
124 $query .= " AND (f.name LIKE :name OR f.common_name LIKE :name)";
125 $params[':name'] = "%{$criteria['name']}%";
126 }
127
128 $query .= " GROUP BY f.id";
129
130 $stmt = $this->pdo->prepare($query);
131 foreach ($params as $key => $value) {
132 $stmt->bindValue($key, $value);
133 }
134 $stmt->execute();
135 return $stmt->fetchAll(PDO::FETCH_ASSOC);
136 }

164 // Obtener hongos similares basados en características
165 public function getSimilarFungi($fungiId, $limit = 5) {
166 $stmt = $this->pdo->prepare("
167 SELECT f2.*, t2.*, c2.*,
168 GROUP_CONCAT(DISTINCT CONCAT(ic.path, i.filename)) as image_urls,
169 (
170 CASE WHEN f1.edibility = f2.edibility THEN 20 ELSE 0 END +
171 CASE WHEN t1.family = t2.family THEN 30 ELSE 0 END +
172 CASE WHEN t1.ordo = t2.ordo THEN 20 ELSE 0 END
173 ) as similarity_score
174 FROM fungi f1
175 JOIN taxonomy t1 ON f1.id = t1.fungi_id
176 JOIN characteristics c1 ON f1.id = c1.fungi_id
177 JOIN fungi f2
178 JOIN taxonomy t2 ON f2.id = t2.fungi_id
179 JOIN characteristics c2 ON f2.id = c2.fungi_id
180 LEFT JOIN fungi_images fi ON f2.id = fi.fungi_id
181 LEFT JOIN images i ON fi.image_id = i.id
182 LEFT JOIN image_config ic ON i.config_key = ic.config_key
183 WHERE f1.id = :id AND f2.id != :id
184 GROUP BY f2.id
185 ORDER BY similarity_score DESC
186 LIMIT :limit
187 ");
188 $stmt->bindParam(':id', $fungiId, PDO::PARAM_INT);
189 $stmt->bindParam(':limit', $limit, PDO::PARAM_INT);
190 $stmt->execute();
191 return $stmt->fetchAll(PDO::FETCH_ASSOC);
