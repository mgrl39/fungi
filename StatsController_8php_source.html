454
455 // Obtener estadísticas de comestibilidad y formatearlas
456 $edibilityStats = $this->getEdibilityStats();
457 $edibilityFormatted = [

463
464 foreach ($edibilityStats as $stat) {

466 if ($stat['edibility'] === 'non_edible') $edibilityFormatted['non_edible_fungi'] = $stat['count'];
467 if ($stat['edibility'] === 'toxic') $edibilityFormatted['toxic_fungi'] = $stat['count'];
468 if ($stat['edibility'] === 'unknown') $edibilityFormatted['unknown_edibility_fungi'] = $stat['count'];
469 }
470
471 // Combinar todos los datos para la plantilla
472 return array_merge($edibilityFormatted, [
473 'total_fungi' => $totalFungi,
474 'total_users' => $this->db->query("SELECT COUNT(*) as total FROM users")->fetch()['total'] ?? 0,
475 'top_families' => $this->getTopFamilies(10),
