171
172 $imageUrls = [];
173

176 if ($imagesResult instanceof \PDOStatement) $images = $imagesResult->fetchAll(\PDO::FETCH_ASSOC);

179 foreach ($images as $image) {

181 $basePath = $this->getImageBasePath($image['config_key']);

192 // Guardar las URLs como una cadena separada por comas (para mantener compatibilidad)
193 $fungus['image_urls'] = implode(',', $imageUrls);
